obj/%.o: src/%.c
	@sleep 0.01
	@mkdir -p obj
	@$(CC) $(CFLAGS) -fPIC -c $< -o $@
	@$(eval COMPILED=$(shell echo $$(($(COMPILED)+1))))
	@$(eval PROGRESS=$(shell echo $$(($(COMPILED)*50/$(TOTAL_FILES)))))
	@printf "\r\e[32mCompiling MINISHELL: [%-50s] %d/%d files\e[0m" \
		$$(printf '█%.0s' $$(seq 1 $(PROGRESS))) $(COMPILED) $(TOTAL_FILES)

$(MINISHELL_LIB): $(OBJ) $(LIBFT)
	@$(AR) $(MINISHELL_LIB) $(OBJ)

$(LIBFT):
	@$(MAKE) $(LIBFT_D)

removep: clean fclean
	@$(RM) $(NAME)
	@$(RM) $(BONUS_NAME)

clean:
	@$(RM) $(OBJ)
	@$(RM) -rf obj
	@make clean -sC $(LIBFT_D)

fclean: clean
	@$(RM) $(MINISHELL_LIB)
	@make fclean -sC $(LIBFT_D)

re: fclean all

.PHONY: all clean fclean re
